<script>
const WORK_MINS = 8 * 60;
const STORAGE_KEY = 'tooaeg_v1';

function pad(n) { return String(n).padStart(2,'0'); }
function nowHHMM() { const d=new Date(); return pad(d.getHours())+':'+pad(d.getMinutes()); }
function todayKey() { const d=new Date(); return `${d.getFullYear()}-${pad(d.getMonth()+1)}-${pad(d.getDate())}`; }
function minsToStr(m) {
  const abs = Math.abs(m);
  const h = Math.floor(abs/60), min = abs%60;
  return (m<0?'-':'')+h+'h '+(min>0?pad(min)+'m':'');
}

function getData() {
  try { return JSON.parse(localStorage.getItem(STORAGE_KEY)) || {}; } catch(e) { return {}; }
}
function saveData(d) { localStorage.setItem(STORAGE_KEY, JSON.stringify(d)); }

function getUserData(name) {
  const d = getData();
  return d[name] || {};
}

function setUserDay(name, key, obj) {
  const d = getData();
  if(!d[name]) d[name] = {};
  d[name][key] = obj;
  saveData(d);
}

let currentUser = null;
let tickInterval = null;

function startSession() {
  const name = document.getElementById('name-input').value.trim();
  if(!name) { document.getElementById('name-input').focus(); return; }
  currentUser = name;
  document.getElementById('setup-screen').style.display = 'none';
  document.getElementById('main-screen').style.display = 'block';
  renderMain();
  tickInterval = setInterval(tick, 10000);
}

function switchUser() {
  clearInterval(tickInterval);
  currentUser = null;
  document.getElementById('setup-screen').style.display = 'block';
  document.getElementById('main-screen').style.display = 'none';
  document.getElementById('name-input').value = '';
}

function getToday() {
  const ud = getUserData(currentUser);
  return ud[todayKey()] || { checkin: null, checkout: null };
}

function setToday(obj) {
  setUserDay(currentUser, todayKey(), obj);
}

function handleMainAction() {
  const today = getToday();
  if(!today.checkin) {
    setToday({ checkin: nowHHMM(), checkout: null });
  } else if(!today.checkout) {
    setToday({ ...today, checkout: nowHHMM() });
  }
  renderMain();
}

function workedMins(ci, co) {
  if(!ci) return 0;
  const end = co || nowHHMM();
  const [ch,cm] = ci.split(':').map(Number);
  const [eh,em] = end.split(':').map(Number);
  return Math.max(0, (eh*60+em) - (ch*60+cm));
}

function renderMain() {
  const today = getToday();
  const d = new Date();
  const days = ['pühapäev','esmaspäev','teisipäev','kolmapäev','neljapäev','reede','laupäev'];
  const months = ['jaanuar','veebruar','märts','aprill','mai','juuni','juuli','august','september','oktoober','november','detsember'];

  document.getElementById('greeting').textContent = 'Tere, ' + currentUser.split(' ')[0] + '!';
  document.getElementById('main-date').textContent = days[d.getDay()] + ', ' + d.getDate() + '. ' + months[d.getMonth()] + ' ' + d.getFullYear();

  document.getElementById('checkin-display').textContent = today.checkin || '—';
  document.getElementById('checkout-display').textContent = today.checkout || '—';

  const wm = workedMins(today.checkin, today.checkout);
  const pct = Math.min(100, Math.round(wm / WORK_MINS * 100));
  const btn = document.getElementById('main-btn');

  if(!today.checkin) {
    btn.textContent = 'Registreeri saabumine';
    btn.className = 'action-btn';
    btn.disabled = false;
    document.getElementById('progress-section').style.display = 'none';
  } else if(!today.checkout) {
    btn.textContent = 'Registreeri lahkumine';
    btn.className = 'action-btn checkout';
    btn.disabled = false;
    document.getElementById('progress-section').style.display = 'block';
    updateProgress(wm, pct, false);
  } else {
    btn.textContent = 'Päev lõpetatud ✓';
    btn.className = 'action-btn';
    btn.disabled = true;
    document.getElementById('progress-section').style.display = 'block';
    updateProgress(wm, pct, true);
  }

  renderHistory();
}

function updateProgress(wm, pct, done) {
  document.getElementById('progress-pct').textContent = pct + '%';
  const fill = document.getElementById('progress-fill');
  fill.style.width = pct + '%';
  fill.className = 'progress-fill' + (pct >= 100 ? ' over' : pct >= 75 ? '' : ' warn');

  const badge = document.getElementById('status-badge-wrap');
  const diff = wm - WORK_MINS;
  let cls, txt;
  if(!done) {
    if(diff >= 0) { cls='badge-ok'; txt='✓ Norm täidetud, ' + minsToStr(diff) + ' üle'; }
    else if(wm >= WORK_MINS*0.75) { cls='badge-warn'; txt='Jääb ' + minsToStr(-diff); }
    else { cls='badge-neutral'; txt=minsToStr(wm) + ' tehtud'; }
  } else {
    if(diff >= 0) { cls='badge-ok'; txt='✓ ' + minsToStr(wm) + ' — norm täidetud'; }
    else { cls='badge-danger'; txt='⚠ Puudub ' + minsToStr(-diff); }
  }
  badge.innerHTML = `<span class="status-badge ${cls}">${txt}</span>`;
}

function renderHistory() {
  const ud = getUserData(currentUser);
  const today = todayKey();
  const entries = Object.entries(ud)
    .filter(([k]) => k !== today)
    .sort((a,b) => b[0].localeCompare(a[0]))
    .slice(0, 10);

  const list = document.getElementById('history-list');
  if(!entries.length) {
    list.innerHTML = '<div class="empty-history">Ajalugu puudub</div>';
    return;
  }

  list.innerHTML = entries.map(([dateKey, rec]) => {
    const wm = rec.checkin && rec.checkout ? workedMins(rec.checkin, rec.checkout) : null;
    const [y,m,dd] = dateKey.split('-');
    const d = new Date(+y,+m-1,+dd);
    const days = ['P','E','T','K','N','R','L'];
    const dateStr = days[d.getDay()] + ' ' + dd + '.' + m;
    const timesStr = rec.checkin ? (rec.checkin + (rec.checkout ? ' – ' + rec.checkout : ' – ?')) : '—';
    const hoursHtml = wm !== null
      ? `<span class="history-hours ${wm >= WORK_MINS ? 'hours-ok' : 'hours-bad'}">${minsToStr(wm)}</span>`
      : `<span class="history-hours" style="color:var(--muted)">—</span>`;
    return `<div class="history-item">
      <span class="history-date">${dateStr}</span>
      <span class="history-times">${timesStr}</span>
      ${hoursHtml}
    </div>`;
  }).join('');
}

function tick() {
  const d = new Date();
  const clock = document.getElementById('live-clock');
  if(clock) clock.textContent = pad(d.getHours()) + ':' + pad(d.getMinutes());
  if(currentUser && getToday().checkin && !getToday().checkout) {
    renderMain();
  }
}

// Header date
const hd = new Date();
document.getElementById('header-date').textContent = hd.toLocaleDateString('et-EE', {weekday:'long', year:'numeric', month:'long', day:'numeric'});

tick();

  const clock = document.getElementById('live-clock');
  if(clock) { const d=new Date(); clock.textContent = pad(d.getHours())+':'+pad(d.getMinutes()); }
}, 10000);
